# Raspberry Pi 4 (BCM2711) boot config for joe2.
# The GPU firmware reads this, then loads kernel8.img to 0x80000 and starts the
# ARM cores. 64-bit is NOT auto-enabled on the Pi 4 — arm_64bit=1 is required.
arm_64bit=1

# Load our raw image as the kernel. (kernel8.img is also the default name the
# firmware looks for in 64-bit mode; named explicitly for clarity.)
kernel=kernel8.img

# Keep firmware from moving the start address; joe2 links everything to 0x80000.
kernel_address=0x80000

# Serial console on the GPIO header, 115200 8N1 via a USB-TTL adapter.
# joe2 outputs on the mini-UART (UART1, GPIO14/15). Pin the VideoCore core clock
# so the mini-UART baud is stable and matches Bcm2711.BAUD_115200 (=541 @ 500MHz).
# The Pi 4 mini-UART tracks the 500 MHz core; the Pi 3 recipe (250 MHz / 270) came
# out ~2x too fast on real silicon and garbled the serial stream.
enable_uart=1
core_freq=500

# No device tree — joe2 talks to the hardware directly, no OS.
device_tree=
disable_commandline_tags=1
